    def initialize(filename, options = {})
      @options = {:in_memory => true, :accessors => true}.merge(options)
      
      @in_memory = @options[:in_memory]
      @accessors = @options[:accessors]
      @data = File.open(filename, 'rb')
      @memo = open_memo(filename)
      reload!
    end
