    def schema(path = nil)
      s = "ActiveRecord::Schema.define do\n"
      s << "  create_table \"#{File.basename(@data.path, ".*")}\" do |t|\n"
      columns.each do |column|
        s << "    t.column \"#{underscore(column.name)}\""
        case column.type
        when "N" # number
          if column.decimal > 0
            s << ", :float"
          else
            s << ", :integer"
          end
        when "D" # date
          s << ", :datetime"
        when "L" # boolean
          s << ", :boolean"
        when "M" # memo
          s << ", :text"
        else
          s << ", :string, :limit => #{column.length}"
        end
        s << "\n"
      end
      s << "  end\nend"
      
      if path
        File.open(path, 'w') {|f| f.puts(s)}
      else
        s
      end
    end
